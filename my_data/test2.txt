# VFS/config.py
"""
Simple Config Loader
"""
from vfs_setup import PROJECT_ROOT
import yaml
from pathlib import Path

def load_config(environment="dev"):
    """Load environment config"""
    config_file = Path(PROJECT_ROOT) / "configs" / f"{environment}.yaml"
    
    if not config_file.exists():
        return None
    
    with open(config_file, 'r') as f:
        config = yaml.safe_load(f)
    
    config["project_root"] = PROJECT_ROOT
    config["environment_name"] = environment
    
    return config

def get_database_name(environment="dev"):
    """Get database name directly"""
    config = load_config(environment)
    return config["database"]["name"] if config else None

def get_storage_account(environment="dev"):
    """Get storage account directly"""
    config = load_config(environment)
    return config["azure"]["storage_account"] if config else None

def get_batch_size(environment="dev"):
    """Get batch size directly"""
    config = load_config(environment)
    return config["processing"]["batch_size"] if config else None
