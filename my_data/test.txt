"""

import sys
from pathlib import Path

# VFS Configuration
VFS_PROJECT_ROOT = "/Workspace/Users/your-email@company.com/VFS"

def ensure_vfs_path():
    """Ensure VFS is in Python path"""
    if VFS_PROJECT_ROOT not in sys.path:
        sys.path.insert(0, VFS_PROJECT_ROOT)
    return VFS_PROJECT_ROOT

def setup_vfs(environment="dev"):
    """
    Setup VFS with specified environment
    
    Args:
        environment (str): Environment to load (dev, prod)
    
    Returns:
        dict: VFS configuration
    """
    # Ensure path
    project_root = ensure_vfs_path()
    
    # Load config
    from vfs_config import load_config
    config = load_config(environment)
    
    if config is None:
        print(f"‚ùå Failed to load environment: {environment}")
        return None
    
    print(f"‚úÖ VFS {environment.upper()} ready!")
    print(f"üìÅ Root: {project_root}")
    print(f"üóÑÔ∏è  Database: {config['database']['name']}")
    
    return {
        'config': config,
        'environment': environment,
        'PROJECT_ROOT': project_root
    }

# Auto-setup when imported (optional)
if __name__ != "__main__":
    ensure_vfs_path()
