SELECT 
  -- Extract PRODUCT NAME (2nd path segment after 'products/')
  REGEXP_EXTRACT(
    file_relative_path, 
    'products/([^/]+)', 
    1
  ) AS product_name,  -- Changed to underscore
  
  -- Display relative path (will convert to hyperlink in Excel)
  file_relative_path AS quick_link_to_research_plan,
  
  -- Full URL for hyperlinking
  CONCAT(
    'https://github.com/department-of-veterans-affairs/va.gov-team/blob/master/',
    TRIM(LEADING '/' FROM file_relative_path)
  ) AS hidden_url
FROM 
  vfs_raw.target_file_metadata
WHERE 
  category = 'research_plan'
ORDER BY 
  product_name
